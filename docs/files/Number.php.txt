namespace Findforsikring\Support;


class Number
{
    public static function formatDanishKroner($amount, $prefix = 'kr. ', $postfix = null)
    {
        return ($prefix ?: '') . number_format($amount, 2, ',', '.') . ($postfix ?: '');
    }
}
